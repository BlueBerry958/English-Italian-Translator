Relazione Traccia - Dizionario Inglese-Italiano

Si vuole simulare la gestione di un dizionario che permette di tradurre un vocabolo dalla lingua Inglese a quella Italiana. Il dizionario è indicizzato tramite le lettere dell’alfabeto e ad ogni vocabolo è associato il corrispondente termine in Italiano. Permettere all’utente di:
• Inserire una frase in Italiano (i termini sono separati da uno spazio). Il programma riconosce i termini e se presenti nel dizionario li traduce in Inglese.
• Effettuare almeno 3 test.

DIZIONARIO.C

C:\Users\Stefano\Desktop\Progetto\C\dizionario.c 
 1 /*
 2   funzione principale del programma (che prende come parametro solo la string di input dell'utente). La sua finalità è: creare il dizionario contenente le parole in inglese e in italiano (in ordine alfabetico),
 3   estrarre i token dalla stringa di input e contare quante parole sono presenti, passare la parola alla funzione che si occupa della traduzione effettiva, analizzare se la traduzione è stata effettuata oppure no;
 4   in questo caso conto quante volte succede e vi è l'attivazione della funzione che si occupa di aggiornare la stringa di parole non tradotte e sostituire nella stringa finale invece della traduzione dei trattini.
 5  
 6   Una volta terminata la stringa di output, attivo una funzione per rimuovere gli spazi in eccesso a fine stringa (sia per la stringa di output sia per quella delle parole non tradotte) e passo alla funzione competente
 7   le variabili di conteggio del numero di parole sia tradotte che non (usate per verificare il tipo di output che il programma ha ottenuto) e la stringa di output e delle parole non traducibili.
 8  */
 9  
10  #include <stdlib.h>
11  #include <string.h>
12  #include "libreria.h"  //libreria contenente l'inizializzazione delle funzioni e delle variabili globali condivise tra le funzioni.
13  
14      //inizializzazione delle variabili globali.
15      int n_parole = 0;  //variabile contatore per il numero di parole DA tradurre.
16      int n_mancate_traduzioni = 0;  //variabile contatore per il numero di parole NON traducibili.
17  
18      char ENG[200] = "";    //Inizializzazione array di output (frase tradotta IN INGLESE).
19      char Parole_Non_Tradotte[200] = "";   //stringa che conterrà tutte le eventuali parole non tradotte.
20  
21  
22  //funzione principale; parametri: ITA = stringa di input da tradurre.
23  void traduttore(char *ITA){
24  
25      Traduzioni Dizionario[] = {     //Dichiaro l'array di "Traduzioni" e inserisco le parole in italiano assieme alle loro traduzioni.
26          {"amico","friend"},         //Le parole sono ordinate in ordine alfabetico.
27          {"benvenuto","welcome"},
28          {"ciao","hello"},
29          {"come","how"},
30          {"hai","did you"},
31          {"inglese","english"},
32          {"mio","of mine"},
33          {"prova","test"},
34          {"stai", "are you"},
35          {"superato","pass"},
36          {"tempo","weather"},
37      };
38  
39      int risultato_ricerca = 0;  //variabile di controllo per il risultato della traduzione di una parola (se è stata tradotta o meno).
40      char *parola;      //Inizializzaione della stringa contenente le parole estratte dall'array ITA (frase da tradurre).
41      parola = strtok(ITA, " ");    //Estraggo la prima parola.
42  
43      while (parola != NULL)  //ciclo finchè non finisco tutte le parole della frase.
44      {
45          risultato_ricerca = Ricerca_Traduzione(parola, Dizionario, sizeof(Dizionario) / sizeof(Dizionario[0])); //aggiorno la stringa di parole tradotte se posso, e aggiorno la variabile di controllo.
46  
47          if(risultato_ricerca == 0)  //controllo se la traduzione è fallita.
48              {
49                  n_mancate_traduzioni += traduzione_fallita(parola);  //aggiorno il numero delle traduzione fallite e aggiorno la lista delle parole non traducibili tramite funzione.
50              }
51  
52          risultato_ricerca = 0;  //reset della variabile per il prossimo ciclo.
53          n_parole++;      //conto quante parole stanno venendo analizzate
54          parola = strtok(NULL, " ");     //Ricavo la parola successiva.
55      }
56  
57      pulizia_stringa();  //funzione per la rimozione degli spazi finali, sia della stringa tradotta sia di quella delle parole non tradotte.
58  
59      output();  //funzione per la stampa dell'output finale (frase tradotta).
60  }

OPERAZIONI.C

C:\Users\Stefano\Desktop\Progetto\C\operazioni.c 
 1 // Funzioni utilizzate dalla funzione principale "dizionario.c".
 2  #include "libreria.h"
 3  #include <stdlib.h>
 4  #include <string.h>
 5  #include <stdio.h>
 6  
 7  //funzione per la traduzione, sfrutta l'algoritmo di ricerca binaria ricorsiva; parametri: p = parola da tradurre, Lista = dizionario contenente le traduzioni, dim = numero di parole.
 8  int Ricerca_Traduzione(char *p, Traduzioni *Lista, int dim) {
 9      int medio;
10  
11      if (dim == 0) {
12          return 0;  //segnala che la traduzione non è stata trovata nel dizionario.
13      }
14  
15      medio = (dim - 1) / 2;
16  
17      if (strcmp(p, Lista[medio].italiano) == 0) {
18          strcat(ENG, Lista[medio].inglese);  //se trova la parola, inserisce la traduzione nella stringa di output.
19          strcat(ENG, " "); //aggiunge uno spazio.
20          return 1;  //segnala alla funzione chiamante che la traduzione è presente.
21      } else if (strcmp(p, Lista[medio].italiano) < 0) {
22          return Ricerca_Traduzione(p, Lista, medio);
23      } else {
24          return Ricerca_Traduzione(p, Lista + 1 + medio, dim - medio - 1);
25      }
26  }
27  
28  
29  //funzione per la creazione della lista di parole non tradotte; p = parola non tradotta.
30  int traduzione_fallita(char *p) {
31      strcat(ENG, "--"); //aggiungi dei tratti per segnalare la traduzione fallita della parola.
32      strcat(ENG, " ");
33  
34      strcat(Parole_Non_Tradotte, p); //aggiungi la parola alla lista di parole non tradotte.
35      strcat(Parole_Non_Tradotte, ", ");
36  
37      return 1; //aumenta il counter delle parole non traducibili.
38  }
39  
40  
41  //funzione per rimuovere gli spazi finali dalle stringhe di output; usa variabili globali.
42  void pulizia_stringa() {
43      int len = strlen(ENG);
44      if (len > 0 && ENG[len - 1] == ' ') {  //rimuovi lo spazio finale e termina la stringa.
45          ENG[len - 1] = '\0';
46      }
47  
48      int len2 = strlen(Parole_Non_Tradotte);
49      if (len2 > 0 && Parole_Non_Tradotte[len2 - 2] == ',') { //rimuovi la virgola in eccesso e lo spazio finale e termina la stringa.
50          Parole_Non_Tradotte[len2 - 2] = '\0';
51      }
52  }
53  
54  
55  //funzione per la stampa dell'output finale; usa variabili globali.
56  void output() {
57      if (n_mancate_traduzioni == n_parole)  //se nessuna parola è stata tradotta.
58          {
59              printf("\nNon e\' stato possibile tradurre la frase (le parole non sono presenti nel dizionario).\n");
60          }
61      else if (n_mancate_traduzioni != 0)  //se solo alcune parole sono state tradotte.
62          {
63              printf("\nFrase tradotta: %s.\n", ENG);
64              printf("\nNon e\' stato possibile tradurre tutte le parole [N:%d - {%s}].\n", n_mancate_traduzioni, Parole_Non_Tradotte);
65          }
66      else  //traduzione completa.
67          {
68              printf("\nFrase tradotta: %s.\n", ENG);
69          }
70  }
